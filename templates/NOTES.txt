     kubectl -n {{ .Release.Namespace }} get svc

     kubectl -n {{ .Release.Namespace }} get ingress fastapi-ing -o yaml | less

Astuce :
  - Logs FastAPI :
    kubectl -n {{ .Release.Namespace }} logs deploy/fastapi -f

  - Postgres :
    kubectl -n {{ .Release.Namespace }} exec -it sts/postgres -- sh -lc 'psql -U "$POSTGRES_USER" -d "$POSTGRES_DB" -c "\l"'
